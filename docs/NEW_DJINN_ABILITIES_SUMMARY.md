## ðŸ”§ **IMPLEMENTATION NOTES**

### **New Ability Properties Added:**
```typescript
// Status effects
statusEffect?: {
  type: 'burn' | 'freeze' | 'stun' | 'paralyze' | 'blind' | 'poison';
  duration: number;
  chance: number; // 0.0-1.0
};

// Drain/siphon
drainPercentage?: number; // Heal for X% of damage dealt

// Stat steal
stealStat?: {
  stat: 'atk' | 'def' | 'mag' | 'spd';
  amount: number;
  duration: number;
};

// Counter damage
counterDamage?: number; // Flat damage
counterDamageAOE?: number; // Damage all attackers

// Heal over time
healOverTime?: {
  amount: number;
  duration: number;
};

// Multi-hit
hitCount?: number;

// Defense penetration
ignoreDefensePercent?: number; // 0.0-1.0

// Splash damage
splashDamagePercent?: number; // 0.0-1.0

// Special protections
invulnerable?: boolean;
immunityStatus?: string[];
immunityAll?: boolean;
physicalImmunity?: boolean;

// Auto-revive
autoRevive?: {
  hpPercent: number;
};

// Revive
revive?: boolean;
reviveHPPercent?: number;

// Elemental
elementalWeakness?: Record<Element, number>;
elementalResistance?: Record<Element, number>;

// Damage reduction
damageReduction?: number;

// Shield
shieldCharges?: number;

// Status removal
removeStatusEffects?: boolean;
```
